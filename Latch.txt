---------------------------LRU CHAIN------------------------------------
SELECT
    name,
    gets,
    misses,
    sleeps,
    ROUND(((gets - misses) / NULLIF(gets, 0)) * 100, 2) AS hit_ratio_percent
FROM
    v$latch
WHERE
    name LIKE '%lru%';
----------------------------LIBRARY CACHE------------------------------
SELECT
    name,
    gets,
    misses,
    sleeps,
    ROUND(((gets - misses) / NULLIF(gets, 0)) * 100, 2) AS hit_ratio_percent
FROM
    v$latch
WHERE
    name LIKE 'library cache%';

----------------------------REDO---------------------------------------
SELECT
    name,
    gets,
    misses,
    sleeps,
    ROUND(((gets - misses) / NULLIF(gets, 0)) * 100, 2) AS hit_ratio_percent
FROM
    v$latch
WHERE
    name LIKE '%redo%'
ORDER BY
    hit_ratio_percent DESC;
